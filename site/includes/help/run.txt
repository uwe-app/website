ht-run 0.18.3
Serve site files

USAGE:
    ht run [OPTIONS] <target>

FLAGS:
        --help       Prints help information
    -V, --version    Prints version information

OPTIONS:
    -h, --host <host>    The name of the host
    -p, --port <port>    The port number

ARGS:
    <target>    Directory to serve files from
